import admin from "firebase-admin";
import fs from "fs";

admin.initializeApp({
  credential: admin.credential.applicationDefault()
});

const db = admin.firestore();

async function backupVentas() {
  const snapshot = await db
    .collection("tenants")
    .doc("Oq1hVXOi1398xZsfabHV")
    .collection("ventas")
    .get();

  if (snapshot.empty) {
    console.log("No hay documentos en ventas");
    return;
  }

  const data = snapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));

  fs.writeFileSync(
    "backup-ventas.json",
    JSON.stringify(data, null, 2)
  );

  console.log("Backup completo de ventas creado âœ”");
}
